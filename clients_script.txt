#!/bin/bash

folder="client_records_500_1024"
n_clients=100

rm ${folder}/*

for ((i=0; i<${n_clients}; i++));
do 
	echo running client Num "$i"
	python3 client.py > "${folder}/C_record_${i}.txt" &
	pids[$i]=$!
	sleep 7s
done

for pid in ${pids[*]}
do
    wait $pid
done

for ((i=0; i<${n_clients}; i++));
do
	cat ${folder}/C_record_${i}.txt
	echo
done
