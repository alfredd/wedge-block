#!/bin/bash

folder="new_client_records_1000_1024"
n_clients=2

rm ${folder}/*

for ((i=0; i<${n_clients}; i++));
do 
	echo running client Num "$i"
	python3 client.py $i > "${folder}/C_record_${i}.txt" &
	pids[$i]=$!
	sleep 15
done

for pid in ${pids[*]}
do
    wait $pid
done
